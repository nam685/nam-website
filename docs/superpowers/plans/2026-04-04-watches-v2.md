```python
import random


def watch_recommended(request):
    """Public: return a weighted-random pinned video for the hero section."""
    visible_tiers = [t for t, _ in WatchChannel.Tier.choices if t != "hidden"]
    tier_weights = {"never_miss": 3, "regular": 2, "check_out": 1}

    pinned_videos = (
        WatchVideo.objects.filter(
            pinned=True,
            visible=True,
            channel__isnull=False,
            channel__tier__in=visible_tiers,
        )
        .select_related("channel")
    )

    if not pinned_videos.exists():
        return JsonResponse({"video": None})

    # Build weighted list
    weighted = []
    for v in pinned_videos:
        weight = tier_weights.get(v.channel.tier, 1)
        weighted.extend([v] * weight)

    chosen = random.choice(weighted)

    return JsonResponse({
        "video": {
            "id": chosen.id,
            "youtube_video_id": chosen.youtube_video_id,
            "title": chosen.title,
            "thumbnail_url": chosen.thumbnail_url,
            "view_count": chosen.view_count,
            "like_count": chosen.like_count,
            "comment_count": chosen.comment_count,
            "description": chosen.description,
            "duration": chosen.duration,
            "channel_name": chosen.channel.name,
            "channel_thumbnail_url": chosen.channel.thumbnail_url,
        }
    })
```

```python
@csrf_exempt
@require_admin
def watch_backfill_stats(_request):
    """Admin: backfill video stats from YouTube API for videos with stale or missing stats."""
    access_token = _refresh_access_token()
    if not access_token:
        return JsonResponse({"error": "YouTube not connected"}, status=400)

    threshold = timezone.now() - timezone.timedelta(days=7)
    stale_videos = WatchVideo.objects.filter(
        models.Q(stats_updated_at__isnull=True) | models.Q(stats_updated_at__lt=threshold)
    )

    video_list = list(stale_videos)
    if not video_list:
        return JsonResponse({"updated": 0})

    updated = 0
    # Batch in groups of 50 (YouTube API limit)
    for i in range(0, len(video_list), 50):
        batch = video_list[i : i + 50]
        video_ids = ",".join(v.youtube_video_id for v in batch)

        try:
            data = _youtube_api_get(
                "videos", access_token,
                {"id": video_ids, "part": "statistics,contentDetails,snippet"},
            )
        except Exception:
            logger.exception("Failed to fetch video stats batch")
            continue

        # Build lookup by video ID
        stats_by_id = {}
        for item in data.get("items", []):
            stats_by_id[item["id"]] = item

        now = timezone.now()
        for v in batch:
            item = stats_by_id.get(v.youtube_video_id)
            if not item:
                continue
            stats = item.get("statistics", {})
            v.view_count = int(stats.get("viewCount", 0))
            v.like_count = int(stats.get("likeCount", 0))
            v.comment_count = int(stats.get("commentCount", 0))
            v.description = item.get("snippet", {}).get("description", "")
            v.duration = item.get("contentDetails", {}).get("duration", "")
            v.stats_updated_at = now
            v.save(update_fields=[
                "view_count", "like_count", "comment_count",
                "description", "duration", "stats_updated_at",
            ])
            updated += 1

    return JsonResponse({"updated": updated})
```

```python
def _sync_liked_videos(access_token, max_pages=4):
    """Paginate YouTube liked videos API, update_or_create WatchVideos, then fetch stats. Returns new count."""
    new_count = 0
    page_token = None
    total = 0
    synced_video_ids = []

    for _ in range(max_pages):
        params = {"myRating": "like", "part": "snippet", "maxResults": 50}
        if page_token:
            params["pageToken"] = page_token

        data = _youtube_api_get("videos", access_token, params)
        items = data.get("items", [])

        for item in items:
            if total >= 200:
                break
            video_id = item.get("id", "")
            if not video_id:
                continue

            snippet = item.get("snippet", {})
            channel_yt_id = snippet.get("channelId", "")
            thumbnails = snippet.get("thumbnails", {})
            thumbnail_url = thumbnails.get("high", thumbnails.get("default", {})).get("url", "")

            # Link to channel if it exists
            channel = None
            if channel_yt_id:
                channel = WatchChannel.objects.filter(youtube_channel_id=channel_yt_id).first()

            _, created = WatchVideo.objects.update_or_create(
                youtube_video_id=video_id,
                defaults={
                    "title": snippet.get("title", ""),
                    "thumbnail_url": thumbnail_url,
                    "channel": channel,
                    "visible": False,
                    "pinned": False,
                },
            )
            if created:
                new_count += 1
            synced_video_ids.append(video_id)
            total += 1

        if total >= 200:
            break

        page_token = data.get("nextPageToken")
        if not page_token:
            break

    # Fetch stats for all synced videos in batches of 50
```

```python
def _fetch_video_stats(access_token, video_ids):
    """Fetch statistics, contentDetails, and snippet for a list of video IDs and update the DB. Returns count updated."""
    now = timezone.now()
    updated = 0
    for i in range(0, len(video_ids), 50):
        batch_ids = video_ids[i : i + 50]
        ids_param = ",".join(batch_ids)
        try:
            data = _youtube_api_get(
                "videos", access_token,
                {"id": ids_param, "part": "statistics,contentDetails,snippet"},
            )
        except Exception:
            logger.exception("Failed to fetch video stats batch")
            continue

        for item in data.get("items", []):
            yt_id = item.get("id", "")
            if not yt_id:
                continue
            try:
                video = WatchVideo.objects.get(youtube_video_id=yt_id)
            except WatchVideo.DoesNotExist:
                continue
            stats = item.get("statistics", {})
            video.view_count = int(stats.get("viewCount", 0))
            video.like_count = int(stats.get("likeCount", 0))
            video.comment_count = int(stats.get("commentCount", 0))
            video.description = item.get("snippet", {}).get("description", "")
            video.duration = item.get("contentDetails", {}).get("duration", "")
            video.stats_updated_at = now
            video.save(update_fields=[
                "view_count", "like_count", "comment_count",
                "description", "duration", "stats_updated_at",
            ])
            updated += 1
    return updated
```

```tsx
  }, []);

  async function handleSync() {
    const token = store("adminToken");
    if (!token) return;
    setSyncing(true);
    try {
      await fetch(`${API}/api/watches/sync/`, {
        method: "POST",
        headers: { Authorization: `Bearer ${token}` },
      });
      await fetchSyncStatus();
      await fetchChannels();
    } finally {
      setSyncing(false);
    }
  }

  function handleConnectYouTube() {
    const token = store("adminToken");
    if (!token) return;
    window.location.href = `${API}/api/watches/auth/?token=${encodeURIComponent(token)}`;
  }

  const handleVideoClick = useCallback((video: WatchVideo, channel: WatchChannel) => {
    setActiveVideo({
      youtube_video_id: video.youtube_video_id,
      title: video.title,
      thumbnail_url: video.thumbnail_url,
      view_count: video.view_count,
      like_count: video.like_count,
      comment_count: video.comment_count,
      description: video.description,
      duration: video.duration,
      channel_name: channel.name,
      channel_thumbnail_url: channel.thumbnail_url,
    });
    setPlaying(true);
    // On mobile, scroll to hero
    if (window.innerWidth < 768 && heroRef.current) {
      heroRef.current.scrollIntoView({ behavior: "smooth" });
    }
  }, []);

  async function handleTierChange(channelId: number, tier: TierKey) {
    const token = store("adminToken");
    if (!token) return;
    const res = await fetch(`${API}/api/watches/channels/${channelId}/tier/`, {
      method: "POST",
      headers: {
        Authorization: `Bearer ${token}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({ tier }),
    });
    if (res.ok) {
      setChannels((prev) =>
        prev.map((ch) => (ch.id === channelId ? { ...ch, tier } : ch)),
      );
    }
  }

  function toggleExpand(id: number) {
    setExpandedId((prev) => (prev === id ? null : id));
  }

  // Build grid items with expansion reflow
  const gridItems = useMemo(() => {
    if (!expandedId) return shuffledChannels.map((ch) => ({ type: "card" as const, channel: ch }));

    const items: ({ type: "card"; channel: WatchChannel } | { type: "expanded"; channel: WatchChannel })[] = [];
    const idx = shuffledChannels.findIndex((ch) => ch.id === expandedId);
    if (idx === -1) return shuffledChannels.map((ch) => ({ type: "card" as const, channel: ch }));

    // Find end of the row containing the clicked card
    const rowEnd = Math.ceil((idx + 1) / COLS) * COLS;

    for (let i = 0; i < shuffledChannels.length; i++) {
      items.push({ type: "card", channel: shuffledChannels[i] });
      // Insert expanded block after the last card in the row
      if (i === Math.min(rowEnd - 1, shuffledChannels.length - 1)) {
        const expandedChannel = shuffledChannels[idx];
        items.push({ type: "expanded", channel: expandedChannel });
      }
    }
    return items;
  }, [shuffledChannels, expandedId, COLS]);

  if (loading) {
    return (
      <div style={{ maxWidth: 900, margin: "0 auto", padding: "2rem 1.5rem", textAlign: "center" }}>
        <p style={{ fontStyle: "italic", color: "#555", fontSize: "0.85rem", fontFamily: "monospace" }}>
          loading...
        </p>
      </div>
    );
  }

  return (
    <>
      <title>Nam watches</title>

      <div
        style={{
          display: "flex",
          maxWidth: 1200,
          margin: "0 auto",
          minHeight: "100vh",
          position: "relative",
          zIndex: 1,
        }}
      >
        {/* Left: Hero Panel (sticky on desktop) */}
        <div
          ref={heroRef}
          style={{
            width: "50%",
            position: "sticky",
            top: 0,
            height: "100vh",
            borderRight: `1px solid ${ACCENT}15`,
            overflowY: "auto",
          }}
          className="watches-hero"
        >
          <HeroPanel
            video={activeVideo}
            playing={playing}
            onPlay={() => setPlaying(true)}
            isAdmin={isAdmin}
            syncStatus={syncStatus}
            syncing={syncing}
            onSync={handleSync}
            onConnectYouTube={handleConnectYouTube}
          />
        </div>

        {/* Right: Scrollable Grid */}
        <div
          style={{
            flex: 1,
            padding: "0.75rem",
            overflowY: "auto",
          }}
        >
          {shuffledChannels.length === 0 ? (
            <p style={{ fontStyle: "italic", color: "#444", fontSize: "0.85rem", textAlign: "center", padding: "3rem 0" }}>
              nothing here yet
            </p>
          ) : (
            <div
              style={{
                display: "grid",
                gridTemplateColumns: "repeat(5, 1fr)",
                gap: "0.4rem",
              }}
              className="watches-grid"
            >
              {gridItems.map((item, i) =>
                item.type === "expanded" ? (
                  <ExpandedCard
                    key={`expanded-${item.channel.id}`}
                    channel={item.channel}
                    onVideoClick={(video) => handleVideoClick(video, item.channel)}
                    isAdmin={isAdmin}
                    onTierChange={handleTierChange}
                  />
                ) : (
                  <ChannelCard
                    key={item.channel.id}
                    channel={item.channel}
                    onClick={() => toggleExpand(item.channel.id)}
                  />
                ),
              )}
            </div>
          )}
        </div>
      </div>

      {/* Responsive styles */}
      <style>{`
        @media (max-width: 1023px) {
          .watches-grid { grid-template-columns: repeat(3, 1fr) !important; }
        }
        @media (max-width: 767px) {
          .watches-hero {
            width: 100% !important;
            position: relative !important;
            height: auto !important;
            border-right: none !important;
            border-bottom: 1px solid ${ACCENT}15;
          }
          .watches-grid { grid-template-columns: repeat(2, 1fr) !important; }
          /* Stack layout on mobile */
        }
        @media (max-width: 767px) {
          .watches-hero + div { width: 100% !important; }
        }
      `}</style>
    </>
  );
}
```
